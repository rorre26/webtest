# pip install aiogram import asyncio from aiogram import Bot, Dispatcher, F from aiogram.types import Message, KeyboardButton, ReplyKeyboardMarkup, WebAppInfo BOT_TOKEN = "ВАШ_ТОКЕН" URL = "https://<ник>.github.io//" bot = Bot(BOT_TOKEN) dp = Dispatcher() @dp.message() async def start(msg: Message): kb = ReplyKeyboardMarkup(resize_keyboard=True, keyboard=[[ KeyboardButton(text="Открыть Mini App", web_app=WebAppInfo(url=URL)) ]]) await msg.answer("Жми кнопку ниже:", reply_markup=kb) @dp.message(F.web_app_data) async def on_webapp_data(msg: Message): await msg.answer(f"Получил из Mini App: {msg.web_app_data.data}") async def main(): await dp.start_polling(bot) if __name__ == "__main__": asyncio.run(main())
